int calculate_sum(int a, int b) {
    return a + b;  
}


void print_array(int arr[], int size) {
    printf("数组元素: ");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}


int factorial(int n) {
    if (n <= 1) {
        return 1;  
    }
    return n * factorial(n - 1);  
}
